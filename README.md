# vue-cli-plugin-ftp-deploy
vue-cli-plugin for deploying via ftp

#initialize

add this to .env

ftpHost = your-ftp-host
ftpPort = your-ftp-port
ftpUsr  = your-ftp-username
ftpPwd  = your-ftp-password

example

ftpHost = 127.0.0.1
ftpPort = 21
ftpUsr  = usr
ftpPwd  = pwdpwdpwd

#usage
edit ftpdeploy/config.js
